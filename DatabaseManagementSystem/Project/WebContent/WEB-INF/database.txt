CREATE TABLE "CITY" ("ID" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL , "NAME" VARCHAR(20) NOT NULL )
CREATE TABLE "LANGUAGE" ("ID" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL , "LANGUAGE" VARCHAR(20) NOT NULL  UNIQUE )
CREATE TABLE "USER" ("ID" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL ,"USER_NAME" VARCHAR(100) NOT NULL UNIQUE ,"PASSWORD" VARCHAR(10) NOT NULL ,"EMAIL" VARCHAR(100) NOT NULL UNIQUE  ,"NAME" VARCHAR(100) NOT NULL ,"SURNAME" VARCHAR(100) NOT NULL ,"LAST_LOGIN" DATETIME,"IS_ACTIVE" BOOL DEFAULT(0),"CONFIRMATION_CODE" VARCHAR(10) NOT NULL , "IS_ADMIN" BOOL DEFAULT(0))
CREATE TABLE "THEATER_TYPE" ("ID" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL , "TYPE" VARCHAR(20) NOT NULL )
CREATE TABLE "ACTOR" ("ID" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,"NAME" VARCHAR(80) NOT NULL ,"SURNAME" VARCHAR(80) NOT NULL ,"VOTE_COUNT" INTEGER NOT NULL  DEFAULT (0) ,"TOTAL_VOTE" INTEGER NOT NULL  DEFAULT (0) )
CREATE TABLE "USER_ACTOR_VOTE" ("ID" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,"USER_ID" INTEGER NOT NULL REFERENCES USER(ID) ON UPDATE CASCADE ON DELETE CASCADE,"ACTOR_ID" INTEGER NOT NULL REFERENCES ACTOR(ID) ON UPDATE CASCADE ON DELETE CASCADE,"VOTE" NUMERIC, CONSTRAINT x UNIQUE (USER_ID, ACTOR_ID))
CREATE TABLE "ACTOR_COMMENT" ("ID" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,"USER_ID" INTEGER NOT NULL REFERENCES USER(ID) ON UPDATE CASCADE ON DELETE CASCADE, "ACTOR_ID" INTEGER NOT NULL REFERENCES ACTOR(ID) ON UPDATE CASCADE ON DELETE CASCADE , "COMMENT" VARCHAR(500) NOT NULL ,"DATE" DATETIME NOT NULL  )
CREATE TABLE "THEATER" ("ID" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,"TITLE" VARCHAR(80) NOT NULL ,"TYPE_ID" INTEGER NOT NULL REFERENCES THEATER_TYPE(ID) ON UPDATE CASCADE ON DELETE RESTRICT ,"LANGUAGE_ID" INTEGER NOT NULL  REFERENCES LANGUAGE(ID) ON UPDATE CASCADE ON DELETE RESTRICT  ,"VOTE_COUNT" INTEGER NOT NULL  DEFAULT (0) ,"TOTAL_VOTE" INTEGER NOT NULL  DEFAULT (0) )
CREATE TABLE "CASTING" (ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL, "THEATER_ID" INTEGER NOT NULL REFERENCES THEATER(ID) ON UPDATE CASCADE ON DELETE CASCADE,"ACTOR_ID" INTEGER NOT NULL REFERENCES ACTOR(ID) ON UPDATE CASCADE ON DELETE RESTRICT,CONSTRAINT x UNIQUE (THEATER_ID, ACTOR_ID))
CREATE TABLE "USER_THEATER_VOTE" ("ID" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,"USER_ID" INTEGER NOT NULL REFERENCES USER(ID) ON UPDATE CASCADE ON DELETE CASCADE,"THEATER_ID" INTEGER NOT NULL REFERENCES THEATER(ID) ON UPDATE CASCADE ON DELETE CASCADE,"VOTE" NUMERIC, CONSTRAINT x UNIQUE (USER_ID, THEATER_ID))
CREATE TABLE "THEATER_COMMENT" ("ID" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,"USER_ID" INTEGER NOT NULL REFERENCES USER(ID) ON UPDATE CASCADE ON DELETE CASCADE, "THEATER_ID" INTEGER NOT NULL REFERENCES THEATER(ID) ON UPDATE CASCADE ON DELETE CASCADE, "COMMENT" VARCHAR(500) NOT NULL ,"DATE" DATETIME NOT NULL  )
CREATE TABLE "EVENT" ("EVENT_NAME" VARCHAR(40) PRIMARY KEY  NOT NULL , "THEATER_ID" INTEGER NOT NULL REFERENCES THEATER(ID) ON UPDATE CASCADE ON DELETE RESTRICT, "CITY_ID" INTEGER NOT NULL REFERENCES CITY(ID) ON UPDATE CASCADE ON DELETE RESTRICT,"LOCATION" VARCHAR(40) NOT NULL ,"DATE" DATETIME NOT NULL  , "SOLD_SEAT" INTEGER NOT NULL  DEFAULT 0, "CAPACITY" INTEGER NOT NULL  DEFAULT 0,CONSTRAINT x UNIQUE (THEATER_ID, DATE))
CREATE TABLE "RESERVATION" ("RESERVATION_CODE" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL ,"EVENT_NAME" VARCHAR(40) NOT NULL REFERENCES EVENT(EVENT_NAME) ON UPDATE RESTRICT ON DELETE RESTRICT ,"USER_ID" INTEGER NOT NULL REFERENCES USER(ID) ON UPDATE CASCADE ON DELETE CASCADE ,"NUM_TICKET" INTEGER NOT NULL )

CREATE TRIGGER TR_ACTOR_VOTE_DELETE BEFORE DELETE ON USER_ACTOR_VOTE BEGIN UPDATE ACTOR SET VOTE_COUNT=VOTE_COUNT-1, TOTAL_VOTE= (TOTAL_VOTE-OLD.VOTE) WHERE ACTOR.ID = OLD.ACTOR_ID; END
CREATE TRIGGER TR_ACTOR_VOTE_UPDATE AFTER UPDATE ON USER_ACTOR_VOTE BEGIN UPDATE ACTOR SET TOTAL_VOTE= (TOTAL_VOTE-OLD.VOTE+NEW.VOTE) WHERE ACTOR.ID = NEW.ACTOR_ID; END
CREATE TRIGGER TR_ACTOR_VOTE_INSERT AFTER INSERT ON USER_ACTOR_VOTE BEGIN UPDATE ACTOR SET VOTE_COUNT = (VOTE_COUNT+1) ,TOTAL_VOTE= (TOTAL_VOTE+NEW.VOTE) WHERE ACTOR.ID = NEW.ACTOR_ID; END

CREATE TRIGGER TR_THEATER_VOTE_INSERT AFTER INSERT ON USER_THEATER_VOTE BEGIN UPDATE THEATER SET VOTE_COUNT = (VOTE_COUNT+1) ,TOTAL_VOTE= (TOTAL_VOTE+NEW.VOTE) WHERE THEATER.ID = NEW.THEATER_ID; END
CREATE TRIGGER TR_THEATER_VOTE_UPDATE AFTER UPDATE ON USER_THEATER_VOTE BEGIN UPDATE THEATER SET TOTAL_VOTE = (TOTAL_VOTE-OLD.VOTE+NEW.VOTE) WHERE THEATER.ID = NEW.THEATER_ID; END
CREATE TRIGGER TR_THEATER_VOTE_DELETE BEFORE DELETE ON USER_THEATER_VOTE BEGIN UPDATE THEATER SET VOTE_COUNT = VOTE_COUNT-1, TOTAL_VOTE= (TOTAL_VOTE-OLD.VOTE) WHERE THEATER.ID = OLD.THEATER_ID; END

CREATE TRIGGER TR_RESERVATION_INSERT AFTER INSERT ON RESERVATION BEGIN UPDATE EVENT SET SOLD_SEAT = (SOLD_SEAT+NEW.NUM_TICKET) WHERE EVENT.EVENT_NAME = NEW.EVENT_NAME; END
CREATE TRIGGER TR_RESERVATION_UPDATE AFTER UPDATE ON RESERVATION BEGIN UPDATE EVENT SET SOLD_SEAT = (SOLD_SEAT-OLD.NUM_TICKET+NEW.NUM_TICKET) WHERE EVENT.EVENT_NAME = NEW.EVENT_NAME; END
CREATE TRIGGER TR_RESERVATION_DELETE BEFORE DELETE ON RESERVATION BEGIN UPDATE EVENT SET SOLD_SEAT = (SOLD_SEAT-OLD.NUM_TICKET) WHERE EVENT.EVENT_NAME = OLD.EVENT_NAME; END

PRAGMA foreign_keys = ON;
